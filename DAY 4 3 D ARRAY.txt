#include <stdio.h>

int main() {
    int arr[2][3][2]; 
    int value = 1;
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 3; j++) {
            for (int k = 0; k < 2; k++) {
                arr[i][j][k] = value++; 
            }
        }
    }
    printf("Elements of the 3D array:\n");
    for (int i = 0; i < 2; i++) {
        printf("Layer %d:\n", i);
        for (int j = 0; j < 3; j++) {
            for (int k = 0; k < 2; k++) {
                printf("%d ", arr[i][j][k]);
            }
            printf("\n");
        }
+        printf("\n");
    }

    return 0;
}
