#include <stdio.h>

int findMax(int arr[], int n) {
    int max = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    return max; 
}
int main() {
    int arr[] = {5, 2, 7, 6, 10, 3}; 
    int n = sizeof(arr) / sizeof(arr[0]); 
    int maximum_element = findMax(arr, n);
    printf("The maximum element in the array is: %d\n", maximum_element);

    return 0;
}
