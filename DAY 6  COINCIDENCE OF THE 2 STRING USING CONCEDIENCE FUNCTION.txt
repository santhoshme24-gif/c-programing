#include <stdio.h>
#include <string.h> // Required for strlen()

// Function to find the coincidence count of two strings
int coincidence(const char *str1, const char *str2) {
    int count = 0;
    int len1 = strlen(str1);
    int len2 = strlen(str2);
    int min_len = (len1 < len2) ? len1 : len2; 
    for (int i = 0; i < min_len; i++) {
        if (str1[i] == str2[i]) {
            count++;
        }
    }

    return count;
}

int main() {
    char string1[100];
    char string2[100];
    int matching_chars;
    printf("Enter the first string: ");
    fgets(string1, sizeof(string1), stdin);
    printf("Enter the second string: ");
    fgets(string2, sizeof(string2), stdin);
    string1[strcspn(string1, "\n")] = '\0';
    string2[strcspn(string2, "\n")] = '\0';
    matching_chars = coincidence(string1, string2);
    printf("\nNumber of coinciding characters: %d\n", matching_chars);

    return 0;
}
