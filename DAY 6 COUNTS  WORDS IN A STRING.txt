#include <stdio.h>
#include <string.h>
#include <ctype.h>

#define MAX_SIZE 100

int main() {
    char str[MAX_SIZE];
    int count = 0;
    int in_word = 0; 

    printf("Enter a string: \n");
    fgets(str, MAX_SIZE, stdin);

    str[strcspn(str, "\n")] = 0;

    for (int i = 0; str[i] != '\0'; i++) {
       
        if (isalnum(str[i]) && in_word == 0) {
            in_word = 1;
            count++;
        }
  
        else if (!isalnum(str[i])) {
            in_word = 0;
        }
    }

    printf("The number of words is: %d\n", count);

    return 0;
}
