#include <stdio.h>
#include <string.h> 

int main() {
    char str[100]; 
    char reversedStr[100];
    int i, j, length;

    printf("Enter a string: ");
  
    fgets(str, sizeof(str), stdin);

    str[strcspn(str, "\n")] = 0;

    length = strlen(str); 
    for (i = length - 1, j = 0; i >= 0; i--, j++) {
        reversedStr[j] = str[i];
    }
    reversedStr[j] = '\0'; 

    printf("Original string: %s\n", str);
    printf("Reversed string: %s\n", reversedStr);

    return 0;
}
