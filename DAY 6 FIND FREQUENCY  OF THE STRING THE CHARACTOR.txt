#include <stdio.h>
#include <string.h> 
int main() {
    char str[1000]; 
    int frequency[256] = {0}; 
    int i;

    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin); 
    str[strcspn(str, "\n")] = 0; 
    for (i = 0; str[i] != '\0'; i++) {
        frequency[(unsigned char)str[i]]++; 
    }

    printf("\nCharacter frequencies:\n");
    for (i = 0; i < 256; i++) {
        if (frequency[i] != 0) {
            printf("Character '%c' appears %d time(s)\n", (char)i, frequency[i]);
        }
    }

    return 0;
}
