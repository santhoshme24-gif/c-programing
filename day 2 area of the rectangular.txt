#include <stdio.h>

int main() {
    float length, width, area;

    // Input length and width
    printf("Enter the length of the rectangle: ");
    scanf("%f", &length);

    printf("Enter the width of the rectangle: ");
    scanf("%f", &width);

    // Calculate area
    area = length * width;

    // Display result
    printf("Area of the rectangle = %.2f\n", area);

    return 0;
}
