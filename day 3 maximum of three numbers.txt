#include <stdio.h>

int main() {
    int num1, num2, num3, max;

    // Input three numbers
    printf("Enter three integers: ");
    scanf("%d %d %d", &num1, &num2, &num3);

    // Find maximum
    max = num1;  // Assume num1 is max initially

    if (num2 > max) {
        max = num2;
    }

    if (num3 > max) {
        max = num3;
    }

    // Display result
    printf("Maximum of %d, %d and %d is %d\n", num1, num2, num3, max);

    return 0;
}
